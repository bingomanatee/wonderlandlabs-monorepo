// Example: Multi-step checkout with validation suspension
store.transact({
  suspendValidation: true, // Allow temporarily invalid states
  action() {
    // Step 1: Clear cart (temporarily invalid - no items but has total)
    this.next({ ...this.value, items: [] })
    
    // Step 2: Process payment (still invalid state)
    this.$.processPayment(this.value.total)
    
    // Step 3: Create order and reset total (now valid)
    this.$.createOrder()
    this.next({ ...this.value, total: 0 }) // Final state is valid
  }
})
