// Primitive values - simple state
const counterForest = new Forest<number>({
  name: 'counter',
  value: 0,
  actions: {
    increment: function() {
      this.next(this.value + 1);
    },
    decrement: function() {
      this.next(this.value - 1);
    },
    reset: function() {
      this.next(0);
    }
  }
});

// String state
const messageForest = new Forest<string>({
  name: 'message',
  value: 'Hello World',
  actions: {
    setMessage: function(value, newMessage: string) {
      this.next(newMessage);
    },
    appendText: function(value, text: string) {
      this.next(value + text);
    }
  }
});

// Boolean state
const toggleForest = new Forest<boolean>({
  name: 'toggle',
  value: false,
  actions: {
    toggle: function() {
      this.next(!this.value);
    },
    setEnabled: function(value, enabled: boolean) {
      this.next(enabled);
    }
  }
});
